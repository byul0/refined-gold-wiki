    function getType(name) {
      const clean = name.replace(/[^a-zA-Z]/g, "");
      return POKEMON_TYPES[clean] || POKEMON_TYPES[name] || "Normal";
    }

    // --- 2. OUTILS POUR LES SPRITES ANIMÉS ---
    function cleanPokemonNameForUrl(name) {
      if (!name) return "";
      return name.toLowerCase()
        .replace(/\s+/g, "")
        .replace(/[^a-z0-9]/g, "")
        .replace("♀", "f")
        .replace("♂", "m");
    }

    function getAnimatedSpriteUrl(name) {
      const cleanName = cleanPokemonNameForUrl(name);
      if (!cleanName) return "";
      return `https://play.pokemonshowdown.com/sprites/ani/${cleanName}.gif`;
    }

    function getPokemonId(name) {
      const clean = cleanPokemonNameForUrl(name);
      return clean; // PokeAPI uses similar clean names for fallback
    }
